pipeline {
    agent any
    stages {
        stage('checkout') {
            steps {
                git "https://github.com/FeliMonti/auto21.git"
            }
        }
        stage('Build') {
            steps {
                sh "mvn compile"
            }
        }
        stage('Test') {
            steps {
                sh "mvn test"
            }
            post {
                always {
                    junit '**/TEST-FooTest.xml'
                }
            }
        }
        stage('Newman') {
            steps {
                sh "newman run Postman_AutoLabb3.postman_collection.json --environment Postman_AutoLabb3_Environment.postman_environment.json --reporters junit"
            }
            post {
                always {
                    junit '**/*xml'
                }
            }
        }
        stage('Code Coverage') {
            steps {
                sh 'mvn clean cobertura:cobertura'
            }
        }
    }
}
